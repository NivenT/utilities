285  static const unordered_set<char> JSON_SYNTAX{
286  '{', '}', '[', ']', ':', ','
287  };
288 
289  queue<JsonToken> tokens;
290  while (!curr.empty()) {
291  JsonToken tkn;
292  if (lex_string(curr, tkn)) {
293  tokens.push(tkn);
294  } else if (lex_number(curr, tkn)) {
295  tokens.push(tkn);
296  } else if (lex_bool(curr, tkn)) {
297  tokens.push(tkn);

299  tokens.push(tkn);

303  tokens.emplace(curr[0]);

313  if (tokens.empty()) return Json::null();

315  tokens.pop();

317  return Json::number(tkn.num);

319  Json obj;
320  while (tokens.front() != JsonToken('}')) {

323  if (tokens.front() != JsonToken(':')) return Json::null();
324  tokens.pop();
325

327  if (tokens.front() == JsonToken(',')) tokens.pop();
328  }

344  return Json::boolean(true);
345  } else if (strcmp(tkn.str, "false") == 0) {
346  return Json::boolean(false);

349  } else {

351  }
